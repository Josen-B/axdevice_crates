25/// Represents the configuration of an emulated device for a virtual machine.
26#[derive(Debug, Default, Clone, serde::Serialize, serde::Deserialize)]
27pub struct EmulatedDeviceConfig {
28    /// The name of the device
29    pub name: String,
30    /// The base IPA (Intermediate Physical Address) of the device.
31    pub base_ipa: usize,
32    /// The length of the device.
33    pub length: usize,
34    /// The IRQ (Interrupt Request) ID of the device.
35    pub irq_id: usize,
36    /// The type of emulated device.
37    pub emu_type: usize,
38    /// The config_list of the device
39    pub cfg_list: Vec<usize>,
40}
41
42/// [`BaseDeviceOps`] is the trait that all emulated devices must implement.
43pub trait BaseDeviceOps<R: DeviceAddrRange> {
44    /// Returns the type of the emulated device.
45    fn emu_type(&self) -> EmuDeviceType;
46    /// Returns the address range of the emulated device.
47    fn address_range(&self) -> R;
48    /// Handles a read operation on the emulated device.
49    fn handle_read(&self, addr: R::Addr, width: AccessWidth) -> AxResult<usize>;
50    /// Handles a write operation on the emulated device.
51    fn handle_write(&self, addr: R::Addr, width: AccessWidth, val: usize) -> AxResult;
52}

54// trait aliases are limited yet: https://github.com/rust-lang/rfcs/pull/3437
55/// [`BaseMmioDeviceOps`] is the trait that all emulated MMIO devices must implement.
56/// It is a trait alias of [`BaseDeviceOps`] with [`GuestPhysAddrRange`] as the address range.
57pub trait BaseMmioDeviceOps = BaseDeviceOps<GuestPhysAddrRange>;
58/// [`BaseSysRegDeviceOps`] is the trait that all emulated system register devices must implement.
59/// It is a trait alias of [`BaseDeviceOps`] with [`SysRegAddrRange`] as the address range.
60pub trait BaseSysRegDeviceOps = BaseDeviceOps<SysRegAddrRange>;
61/// [`BasePortDeviceOps`] is the trait that all emulated port devices must implement.
62/// It is a trait alias of [`BaseDeviceOps`] with [`PortRange`] as the address range.
63pub trait BasePortDeviceOps = BaseDeviceOps<PortRange>;
