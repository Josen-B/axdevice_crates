1use core::fmt::{Display, Formatter};

3/// Enumeration representing the type of emulator devices.
4#[derive(Clone, Copy, Debug, PartialEq, serde::Serialize, serde::Deserialize)]
5#[repr(usize)]
6pub enum EmuDeviceType {
7    /// Console device.
8    EmuDeviceTConsole = 0,
9    /// Interrupt controller device, e.g. vGICv2 in aarch64, vLAPIC in x86.
10    EmuDeviceTInterruptController = 1,
11    /// Partial passthrough interrupt controller device.
12    EmuDeviceTGPPT = 2,
13    /// Virtio block device.
14    EmuDeviceTVirtioBlk = 3,
15    /// Virtio net device.
16    EmuDeviceTVirtioNet = 4,
17    /// Virtio console device.
18    EmuDeviceTVirtioConsole = 5,
19    /// IOMMU device.
20    EmuDeviceTIOMMU = 6,
21    /// Interrupt ICC SRE device.
22    EmuDeviceTICCSRE = 7,
23    /// Interrupt ICC SGIR device.
24    EmuDeviceTSGIR = 8,
25    /// Interrupt controller GICR device.
26    EmuDeviceTGICR = 9,
27    /// A emulated device that provides Inter-VM Communication (IVC) channel.
28    /// This device is used for communication between different VMs,
29    /// the corresponding memory region of this device should be marked as `Reserved` in
30    /// device tree or ACPI table.
31    EmuDeviceTIVCChannel = 10,
32    /// Meta device.
33    EmuDeviceTMeta = 11,
34}
35
36impl Default for EmuDeviceType {
37    fn default() -> Self {
38        Self::EmuDeviceTMeta
39    }
40}
41
42impl Display for EmuDeviceType {
43    // Implementation of the Display trait for EmuDeviceType.
44    fn fmt(&self, f: &mut Formatter<'_>) -> core::fmt::Result {
45        match self {
46            EmuDeviceType::EmuDeviceTConsole => write!(f, "console"),
47            EmuDeviceType::EmuDeviceTInterruptController => write!(f, "Interrupt controller"),
48            EmuDeviceType::EmuDeviceTGPPT => {
49                write!(f, "partial passthrough interrupt controller")
50            }
51            EmuDeviceType::EmuDeviceTVirtioBlk => write!(f, "virtio block"),
52            EmuDeviceType::EmuDeviceTVirtioNet => write!(f, "virtio net"),
53            EmuDeviceType::EmuDeviceTVirtioConsole => write!(f, "virtio console"),
54            EmuDeviceType::EmuDeviceTIOMMU => write!(f, "IOMMU"),
55            EmuDeviceType::EmuDeviceTICCSRE => write!(f, "interrupt ICC SRE"),
56            EmuDeviceType::EmuDeviceTSGIR => write!(f, "interrupt ICC SGIR"),
57            EmuDeviceType::EmuDeviceTGICR => write!(f, "interrupt controller gicr"),
58            EmuDeviceType::EmuDeviceTIVCChannel => write!(f, "IVC channel"),
59            EmuDeviceType::EmuDeviceTMeta => write!(f, "meta device"),
60        }
61    }
62}
63
64/// Implementation of methods for EmuDeviceType.
65impl EmuDeviceType {
66    /// Returns true if the device is removable.
67    pub fn removable(&self) -> bool {
68        matches!(
69            *self,
70            EmuDeviceType::EmuDeviceTInterruptController
71                | EmuDeviceType::EmuDeviceTSGIR
72                | EmuDeviceType::EmuDeviceTICCSRE
73                | EmuDeviceType::EmuDeviceTGPPT
74                | EmuDeviceType::EmuDeviceTVirtioBlk
75                | EmuDeviceType::EmuDeviceTVirtioNet
76                | EmuDeviceType::EmuDeviceTGICR
77                | EmuDeviceType::EmuDeviceTVirtioConsole
78        )
79    }
80
81    /// Converts a usize value to an EmuDeviceType.
82    pub fn from_usize(value: usize) -> EmuDeviceType {
83        match value {
84            0 => EmuDeviceType::EmuDeviceTConsole,
85            1 => EmuDeviceType::EmuDeviceTInterruptController,
86            2 => EmuDeviceType::EmuDeviceTGPPT,
87            3 => EmuDeviceType::EmuDeviceTVirtioBlk,
88            4 => EmuDeviceType::EmuDeviceTVirtioNet,
89            5 => EmuDeviceType::EmuDeviceTVirtioConsole,
90            6 => EmuDeviceType::EmuDeviceTIOMMU,
91            7 => EmuDeviceType::EmuDeviceTICCSRE,
92            8 => EmuDeviceType::EmuDeviceTSGIR,
93            9 => EmuDeviceType::EmuDeviceTGICR,
94            10 => EmuDeviceType::EmuDeviceTIVCChannel,
95            11 => EmuDeviceType::EmuDeviceTMeta,
96            _ => panic!("Unknown  EmuDeviceType value: {}", value),
97        }
98    }
